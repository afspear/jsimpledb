006package org.jsimpledb.kv.mvcc;
007
008import org.jsimpledb.kv.CloseableKVStore;
009import org.jsimpledb.kv.KVStore;
010
011/**
012 * Extension of the {@link KVStore} interface for implementations that support atomic, batched reads and writes.
013 *
014 * <p>
015 * Atomic batch reads are available via {@link #snapshot}, which returns a consistent point-in-time views of the {@link KVStore}.
016 * Atomic batch writes are available via {@link #mutate mutate()}, which applies a set of mutations in an "all or none" fashion.
017 */
018public interface AtomicKVStore extends KVStore {
019
020    /**
021     * Acquire a read-only, snapshot view of this instance based on the current state.
022     *
023     * <p>
024     * The returned {@link KVStore} view should remain constant even if this instance is subsequently mutated.
025     *
026     * <p>
027     * Note: callers are required to {@link CloseableKVStore#close close} the returned instance when no longer in use.
028     *
029     * @return read-only, snapshot view of this instance
030     * @throws org.jsimpledb.kv.StaleTransactionException if an underlying transaction is no longer usable
031     * @throws org.jsimpledb.kv.RetryTransactionException if an underlying transaction must be retried and is no longer usable
032     */
033    CloseableKVStore snapshot();
034
035    /**
036     * Apply a set of mutations to this instance atomically.
037     *
038     * <p>
039     * If this method returns normally, all of the given mutations will have been applied. If this method
040     * returns abnormally, then none of the given mutations will have been applied.
041     *
042     * <p>
043     * In any case, other threads observing this instance will never see a partial application of the given mutations.
044     *
045     * <p>
046     * This method is required to apply the mutations in this order: removes, puts, adjusts.
047     *
048     * <p>
049     * If {@code sync} is true, the implementation must durably persist the changes before returning.
050     *
051     * @param mutations the mutations to apply
052     * @param sync if true, caller requires that the changes be durably persisted
053     * @throws org.jsimpledb.kv.StaleTransactionException if an underlying transaction is no longer usable
054     * @throws org.jsimpledb.kv.RetryTransactionException if an underlying transaction must be retried and is no longer usable
055     * @throws UnsupportedOperationException if {@code sync} is true and this implementation cannot guarantee durability
056     * @throws IllegalArgumentException if {@code mutations} is null
057     */
058    void mutate(Mutations mutations, boolean sync);
059}
