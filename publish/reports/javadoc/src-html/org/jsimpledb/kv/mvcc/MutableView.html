025/**
026 * Provides a mutable view of an underlying, read-only {@link KVStore}.
027 *
028 * <p>
029 * Instances intercept all operations to the underlying {@link KVStore}, recording mutations in a {@link Writes} instance
030 * instead of applying them to the {@link KVStore}. Instances then provide a view of the mutated {@link KVStore} based those
031 * mutations. Mutations that overwrite previous mutations are consolidated.
032 * </p>
033 *
034 * <p>
035 * Unlike writes, reads are passed through to the underlying {@link KVStore}, except where they intersect a previous write.
036 * Reads may also be optionally recorded.
037 * </p>
038 *
039 * <p>
040 * In all cases, then underlying {@link KVStore} is never modified.
041 * </p>
042 *
043 * <p>
044 * Instances are also capable of performing certain MVCC-related calculations, such as whether two transactions may be re-ordered.
045 * </p>
046 *
047 * <p>
048 * Instances are thread safe; however, directly accessing the associated {@link Reads} or {@link Writes} is not thread safe.
049 * </p>
050 */
051public class MutableView extends AbstractKVStore implements SizeEstimating {
052
053    private final KVStore kv;
054    private final Writes writes;
055    private Reads reads;
056
057// Constructors
058
059    /**
060     * Constructor.
061     *
062     * @param kv underlying {@link KVStore}
063     * @throws IllegalArgumentException if {@code kv} is null
064     */
065    public MutableView(KVStore kv) {
066        this(kv, new Reads(), new Writes());
067    }
068
069    /**
070     * Constructor using caller-provided {@link Reads} (optional} and {@link Writes}.
071     *
072     * @param kv underlying {@link KVStore}
073     * @param reads recorded reads, or null for none
074     * @param writes recorded writes
075     * @throws IllegalArgumentException if {@code kv} is null
076     * @throws IllegalArgumentException if {@code writes} is null
077     */
078    public MutableView(KVStore kv, Reads reads, Writes writes) {
079        Preconditions.checkArgument(kv != null, "null kv");
080        Preconditions.checkArgument(writes != null, "null writes");
081        this.kv = kv;
082        this.reads = reads;
083        this.writes = writes;
084        synchronized (this) { }                                     // because this.reads is not final
085    }

121// KVStore

123    @Override
124    public synchronized byte[] get(byte[] key) {
125
126        // Sanity check
127        assert this.check();
128
129        // Check puts
130        final byte[] putValue = this.writes.getPuts().get(key);
131        if (putValue != null)
132            return putValue;
133
134        // Check removes
135        if (this.writes.getRemoves().contains(key))
136            return null;
137
138        // Read from k/v store
139        byte[] value = this.kv.get(key);
140
141        // Record the read
142        this.recordReads(key, ByteUtil.getNextKey(key));
143
144        // Check counter adjustments
145        if (value == null)                      // we can ignore adjustments of missing values
146            return null;
147        final Long adjust = this.writes.getAdjusts().get(key);
148        if (adjust != null) {
149
150            // Decode value we just read as a counter
151            final long counterValue;
152            try {
153                counterValue = this.kv.decodeCounter(value);
154            } catch (IllegalArgumentException e) {
155                this.writes.getAdjusts().remove(key);       // previous adjustment was bogus because value was not decodable
156                return value;
157            }
158
159            // Adjust counter value by adjustment and re-encode
160            value = this.kv.encodeCounter(counterValue + adjust);
161        }
162
163        // Done
164        return value;
165    }
166
167    @Override
168    public Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse) {
169
170        // Sanity check
171        assert this.check();
172
173        // Build iterator
174        return new RangeIterator(minKey, maxKey, reverse);
175    }
176
177    @Override
178    public synchronized void put(byte[] key, byte[] value) {
179
180        // Sanity check
181        assert this.check();
182        Preconditions.checkArgument(key != null, "null key");
183        Preconditions.checkArgument(value != null, "null value");
184
185        // Overwrite any counter adjustment
186        this.writes.getAdjusts().remove(key);
187
188        // Overwrite any removal
189        if (this.writes.getRemoves().contains(key))
190            this.writes.setRemoves(this.writes.getRemoves().remove(new KeyRange(key)));
191
192        // Record the put
193        this.writes.getPuts().put(key.clone(), value.clone());
194    }
195
196    @Override
197    public synchronized void remove(byte[] key) {
198
199        // Sanity check
200        assert this.check();
201        Preconditions.checkArgument(key != null, "null key");
202
203        // Overwrite any counter adjustment
204        this.writes.getAdjusts().remove(key);
205
206        // Overwrite any put
207        this.writes.getPuts().remove(key);
208
209        // Record the remove
210        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(key)));
211    }
212
213    @Override
214    public synchronized void removeRange(byte[] minKey, byte[] maxKey) {
215
216        // Sanity check
217        assert this.check();
218
219        // Realize minKey
220        if (minKey == null)
221            minKey = ByteUtil.EMPTY;
222
223        // Overwrite any puts and counter adjustments
224        if (maxKey != null) {
225            this.writes.getPuts().subMap(minKey, maxKey).clear();
226            this.writes.getAdjusts().subMap(minKey, maxKey).clear();
227        } else {
228            this.writes.getPuts().tailMap(minKey).clear();
229            this.writes.getAdjusts().tailMap(minKey).clear();
230        }
231
232        // Record the remove
233        this.writes.setRemoves(this.writes.getRemoves().add(new KeyRange(minKey, maxKey)));
234    }

246    @Override
247    public synchronized void adjustCounter(byte[] key, long amount) {
248
249        // Sanity check
250        assert this.check();
251
252        // Check puts
253        final byte[] putValue = this.writes.getPuts().get(key);
254        if (putValue != null) {
255            final long value;
256            try {
257                value = this.kv.decodeCounter(putValue);
258            } catch (IllegalArgumentException e) {
259                return;                                 // previously put value was not decodable, so ignore this adjustment
260            }
261            this.writes.getPuts().put(key, this.kv.encodeCounter(value + amount));
262            return;
263        }
264
265        // Check removes
266        if (this.writes.getRemoves().contains(key))
267            return;
268
269        // Calculate new, cumulative adjustment
270        final Long oldAdjust = this.writes.getAdjusts().get(key);
271        final long adjust = (oldAdjust != null ? oldAdjust : 0) + amount;
272
273        // Record/update adjustment
274        if (adjust != 0)
275            this.writes.getAdjusts().put(key, adjust);
276        else
277            this.writes.getAdjusts().remove(key);
278    }

309// Internal methods

311    // Record that keys were read in the range [minKey, maxKey)
312    private synchronized void recordReads(byte[] minKey, byte[] maxKey) {
313
314        // Not tracking reads?
315        if (this.reads == null)
316            return;
317
318        // Realize minKey
319        if (minKey == null)
320            minKey = ByteUtil.EMPTY;
321
322        // Already tracked?
323        final KeyRange readRange = new KeyRange(minKey, maxKey);
324        if (this.reads.getReads().contains(readRange))
325            return;
326
327        // Subtract out the part of the read range that did not really go through to k/v store due to puts or removes
328        KeyRanges readRanges = new KeyRanges(readRange);
329        final Set<byte[]> putKeys = (maxKey != null ?
330          this.writes.getPuts().subMap(minKey, maxKey) : this.writes.getPuts().tailMap(minKey)).keySet();
331        for (byte[] key : putKeys)
332            readRanges = readRanges.remove(new KeyRange(key));
333        readRanges = readRanges.intersection(this.writes.getRemoves().inverse());
334
335        // Record reads
336        if (!readRanges.isEmpty())
337            this.reads.setReads(this.reads.getReads().union(readRanges));
338    }

340// Debugging
341
342    // Verify puts, removes, and adjusts are all mutually disjoint
343    private synchronized boolean check() {
344        MutableView.verifyDisjoint(
345          this.writes.getRemoves(),
346          MutableView.buildKeyRanges(this.writes.getPuts().keySet()),
347          MutableView.buildKeyRanges(this.writes.getAdjusts().keySet()));
348        return true;
349    }
350
351    private static void verifyDisjoint(KeyRanges... rangeses) {
352        for (int i = 0; i < rangeses.length - 1; i++) {
353            for (int j = i + 1; j < rangeses.length; j++)
354                assert rangeses[i].intersection(rangeses[j]).isEmpty();
355        }
356    }
357
358    private static KeyRanges buildKeyRanges(Iterable<byte[]> keys) {
359        KeyRanges ranges = KeyRanges.EMPTY;
360        for (byte[] key : keys)
361            ranges = ranges.add(new KeyRange(key));
362        return ranges;
363    }

365// RangeIterator

367    private class RangeIterator implements Iterator<KVPair> {
368
369        private final KVPairIterator pi;
370        private final byte[] limit;
371        private final boolean reverse;
372
373        private byte[] cursor;                  // current position; inclusive if forward, exclusive if reverse
374        private KVPair next;                    // the next k/v pair queued up, or null if not found yet
375        private byte[] removeKey;               // key to remove if remove() is invoked
376        private boolean finished;
377
378        RangeIterator(byte[] minKey, byte[] maxKey, boolean reverse) {
379
380            // Realize minKey
381            if (minKey == null)
382                minKey = ByteUtil.EMPTY;
383
384            // Build KVPairIterator that omits keys we've put or removed so far; this is safe even if more keys are put and/or
385            // removed after creation, because the set "keys we've put or removed so far" can only increase over time.
386            synchronized (MutableView.this) {
387                final KeyRanges putsAndRemoves = MutableView.buildKeyRanges(MutableView.this.writes.getPuts().keySet())
388                  .union(MutableView.this.writes.getRemoves());
389                this.pi = new KVPairIterator(MutableView.this.kv, new KeyRange(minKey, maxKey), putsAndRemoves.inverse(), reverse);
390            }
391
392            // Initialize cursor
393            this.cursor = reverse ? maxKey : minKey;
394            this.limit = reverse ? minKey : maxKey;
395            this.reverse = reverse;
396        }
397
398        @Override
399        public synchronized boolean hasNext() {
400            return this.next != null || this.findNext();
401        }
402
403        @Override
404        public synchronized KVPair next() {
405            if (this.next == null && !this.findNext())
406                throw new NoSuchElementException();
407            final KVPair pair = this.next;
408            assert pair != null;
409            this.removeKey = pair.getKey();
410            this.next = null;
411            return pair;
412        }
413
414        @Override
415        public synchronized void remove() {
416            Preconditions.checkState(this.removeKey != null);
417            MutableView.this.remove(this.removeKey);
418            this.removeKey = null;
419        }
420
421        private synchronized boolean findNext() {
422
423            // Invariants
424            assert this.next == null;
425            assert this.cursor != null || this.reverse;
426
427            // Exhausted?
428            if (this.finished)
429                return false;
430
431            // Compare against most current MutableView puts and removes
432            synchronized (MutableView.this) {
433
434                // Find the next k/v pair not filtered out by (newly added) removes, if any
435                KVPair readPair = null;
436                while (this.pi.hasNext()) {
437                    readPair = this.pi.next();
438                    if (MutableView.this.writes.getRemoves().contains(readPair.getKey())) {
439                        readPair = null;
440                        continue;
441                    }
442                    break;
443                }
444
445                // Find the next put, if any
446                final Map.Entry<byte[], byte[]> putEntry = this.reverse ?
447                  (this.cursor != null ?
448                   MutableView.this.writes.getPuts().lowerEntry(this.cursor) : MutableView.this.writes.getPuts().lastEntry()) :
449                  MutableView.this.writes.getPuts().ceilingEntry(this.cursor);
450                final KVPair putPair = putEntry != null ? new KVPair(putEntry) : null;
451
452                // Figure out which pair wins (read or put)
453                final KVPair pair;
454                int diff = 0;
455                if (readPair == null && putPair == null)
456                    pair = null;
457                else if (readPair == null)
458                    pair = putPair;
459                else if (putPair == null)
460                    pair = readPair;
461                else {
462                    diff = ByteUtil.compare(putPair.getKey(), readPair.getKey());
463                    if (reverse)
464                        diff = -diff;
465                    pair = diff <= 0 ? putPair : readPair;                  // if there's a tie, the put wins
466                }
467
468                // Record that we read from everything we just scanned over in the underlying KVStore
469                final byte[] skipMin;
470                final byte[] skipMax;
471                if (this.reverse) {
472                    skipMin = pair != null ? pair.getKey() : ByteUtil.EMPTY;
473                    skipMax = this.cursor;
474                } else {
475                    skipMin = this.cursor;
476                    skipMax = pair != null ? ByteUtil.getNextKey(pair.getKey()) : null;
477                }
478                if (skipMax == null || ByteUtil.compare(skipMin, skipMax) < 0)
479                    MutableView.this.recordReads(skipMin, skipMax);
480
481                // Finished?
482                if (pair == null) {
483                    this.finished = true;
484                    return false;
485                }
486
487                // Update state
488                this.next = pair;
489                this.cursor = pair != null ? (this.reverse ? pair.getKey() : ByteUtil.getNextKey(pair.getKey())) : null;
490
491                // If a put appeared prior to the KVPPairIterator's next k/v pair, backup the KVPairIterator for next time
492                if (diff < 0)
493                    this.pi.setNextTarget(this.cursor);
494
495                // Done
496                return this.next != null;
497            }
498        }
499    }
