018/**
019 * A {@link Console} connected to a telnet client via a {@link Socket}.
020 *
021 * <p>
022 * This class can be used to implement an embedded CLI console within an application, accessed via telnet. For example:
023 * <pre>
024 *      // Accept a new connection
025 *      final Socket socket = ...
026 *
027 *      // Set up telnet CLI console
028 *      final TelnetConsole console = TelnetConsole.create(database, socket);
029 *      console.getSession().registerStandardFunctions();
030 *      console.getSession().registerStandardCommands();
031 *
032 *      // Run the console
033 *      console.run();
034 * </pre>
035 */
036public final class TelnetConsole extends Console {
037
038    /**
039     * Internal constructor.
040     */
041    private TelnetConsole(Database db, JSimpleDB jdb, InputStream input, OutputStream output,
042      jline.Terminal terminal, String encoding, String appName) throws IOException {
043        super(db, jdb, input, output, terminal, encoding, appName);
044    }
045
046    /**
047     * Simplified factory method for core API CLI mode.
048     *
049     * @param db core API {@link Database}
050     * @param socket socket connected to telnet client
051     * @return new telnet console instance
052     * @throws IOException if an I/O error occurs
053     */
054    public static TelnetConsole create(Database db, Socket socket) throws IOException {
055        return TelnetConsole.create(db, null, socket.getInputStream(), socket.getOutputStream(), null, null);
056    }
057
058    /**
059     * Simplified factory method for {@link JSimpleDB} API CLI mode.
060     *
061     * @param jdb {@link JSimpleDB} database
062     * @param socket socket connected to telnet client
063     * @return new telnet console instance
064     * @throws IOException if an I/O error occurs
065     */
066    public static TelnetConsole create(JSimpleDB jdb, Socket socket) throws IOException {
067        return TelnetConsole.create(null, jdb, socket.getInputStream(), socket.getOutputStream(), null, null);
068    }
069
070    /**
071     * Generic factory method.
072     *
073     * @param db core API {@link Database}; must be null if and only if {@code jdb} is not null
074     * @param jdb {@link JSimpleDB} database; must be null if and only if {@code db} is not null
075     * @param input console input
076     * @param output console output
077     * @param encoding character encoding for {@code terminal}, or null for default
078     * @param appName JLine application name, or null for none
079     * @return new telnet console instance
080     * @throws IOException if an I/O error occurs
081     * @throws IllegalArgumentException if {@code db} and {@code jdb} are both null or both not null
082     */
083    public static TelnetConsole create(Database db, JSimpleDB jdb, InputStream input, OutputStream output,
084      String encoding, String appName) throws IOException {
085
086        // Set up nvt4j; ignore the initial clear & reposition
087        final nvt4j.impl.Terminal nvt4jTerminal = new nvt4j.impl.Terminal(input, output) {
088
089            private boolean cleared;
090            private boolean moved;
091
092            @Override
093            public void clear() throws IOException {
094                if (this.cleared)
095                    super.clear();
096                this.cleared = true;
097            }
098
099            @Override
100            public void move(int row, int col) throws IOException {
101                if (this.moved)
102                    super.move(row, col);
103                this.moved = true;
104            }
105        };
106        nvt4jTerminal.put(nvt4j.impl.Terminal.AUTO_WRAP_ON);
107        nvt4jTerminal.setCursor(true);
108
109        // Have JLine do input & output through telnet terminal
110        final InputStream jlineInput = new InputStream() {
111            @Override
112            public int read() throws IOException {
113                return nvt4jTerminal.get();
114            }
115        };
116        final OutputStream jlineOutput = new OutputStream() {
117            @Override
118            public void write(int value) throws IOException {
119                nvt4jTerminal.put(value);
120            }
121        };
122
123        // Build console
124        return new TelnetConsole(db, jdb, jlineInput, jlineOutput,
125          new TelnetTerminal(TerminalFactory.get(), nvt4jTerminal), encoding, appName);
126    }

130    /**
131     * Wrapper class for a JLine {@link jline.Terminal} that forwards all methods to the wrapped delegate.
132     */
133    public static class ForwardingTerminal implements jline.Terminal {
134
135        protected final jline.Terminal terminal;
136
137        public ForwardingTerminal(jline.Terminal terminal) {
138            if (terminal == null)
139                throw new IllegalArgumentException("null terminal");
140            this.terminal = terminal;
141        }
142
143        @Override
144        public void init() throws Exception {
145            this.terminal.init();
146        }
147
148        @Override
149        public void restore() throws Exception {
150            this.terminal.restore();
151        }
152
153        @Override
154        public void reset() throws Exception {
155            this.terminal.reset();
156        }
157
158        @Override
159        public boolean isSupported() {
160            return this.terminal.isSupported();
161        }
162
163        @Override
164        public int getWidth() {
165            return this.terminal.getWidth();
166        }
167
168        @Override
169        public int getHeight() {
170            return this.terminal.getHeight();
171        }
172
173        @Override
174        public boolean isAnsiSupported() {
175            return this.terminal.isAnsiSupported();
176        }
177
178        @Override
179        public OutputStream wrapOutIfNeeded(OutputStream out) {
180            return this.terminal.wrapOutIfNeeded(out);
181        }
182
183        @Override
184        public InputStream wrapInIfNeeded(InputStream in) throws IOException {
185            return this.terminal.wrapInIfNeeded(in);
186        }
187
188        @Override
189        public boolean hasWeirdWrap() {
190            return this.terminal.hasWeirdWrap();
191        }
192
193        @Override
194        public boolean isEchoEnabled() {
195            return this.terminal.isEchoEnabled();
196        }
197
198        @Override
199        public void setEchoEnabled(boolean enabled) {
200            this.terminal.setEchoEnabled(enabled);
201        }
202
203        @Override
204        public String getOutputEncoding() {
205            return this.terminal.getOutputEncoding();
206        }
207    }

209// TelnetTerminal

211    /**
212     * JLine {@link jline.Terminal} that gets window size information from an associated nvt4j {@link nvt4j.impl.Terminal}.
213     */
214    public static class TelnetTerminal extends ForwardingTerminal {
215
216        private final nvt4j.impl.Terminal nvt;
217
218        public TelnetTerminal(jline.Terminal terminal, nvt4j.impl.Terminal nvt) {
219            super(terminal);
220            if (nvt == null)
221                throw new IllegalArgumentException("null nvt");
222            this.nvt = nvt;
223        }
224
225        @Override
226        public int getWidth() {
227            final int width = this.nvt.getColumns();
228            return width > 0 ? width : super.getWidth();
229        }
230
231        @Override
232        public int getHeight() {
233            final int height = this.nvt.getRows();
234            return height > 0 ? height : super.getHeight();
235        }
236
237        // TODO: echoEnabled <-> EchoOptionHandler, possibly others
238    }
