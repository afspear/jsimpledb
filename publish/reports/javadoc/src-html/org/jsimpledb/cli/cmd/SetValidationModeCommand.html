006package org.jsimpledb.cli.cmd;
007
008import java.util.Map;
009
010import org.jsimpledb.ValidationMode;
011import org.jsimpledb.cli.CliSession;
012import org.jsimpledb.parse.EnumNameParser;
013import org.jsimpledb.parse.ParseContext;
014import org.jsimpledb.parse.Parser;
015
016@Command
017public class SetValidationModeCommand extends AbstractCommand {
018
019    public SetValidationModeCommand() {
020        super("set-validation-mode mode:mode");
021    }
022
023    @Override
024    public String getHelpSummary() {
025        return "Sets the validation mode for JSimpleDB transactions";
026    }
027
028    @Override
029    public String getHelpDetail() {
030        return "Sets the validation mode for JSimpleDB transactions, one of `disabled', `manual', or `automatic' (the default)."
031          + " This setting is only used when the CLI is running in JSimpleDB mode.";
032    }
033
034    @Override
035    protected Parser<?> getParser(String typeName) {
036        return "mode".equals(typeName) ? new EnumNameParser<ValidationMode>(ValidationMode.class) : super.getParser(typeName);
037    }
038
039    @Override
040    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
041        final ValidationMode mode = (ValidationMode)params.get("mode");
042        return new CliSession.Action() {
043            @Override
044            public void run(CliSession session) throws Exception {
045                session.setValidationMode(mode);
046                session.getWriter().println("Set validation mode to " + mode);
047                if (session.getJSimpleDB() == null)
048                    session.getWriter().println("Warning: no JSimpleDB instance was created, so validation mode will not be used");
049            }
050        };
051    }
052}
