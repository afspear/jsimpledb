006package org.jsimpledb.kv.mvcc;
007
008import java.util.ArrayList;
009import java.util.Iterator;
010import java.util.Map;
011import java.util.TreeMap;
012
013import org.jsimpledb.kv.KVDatabase;
014import org.jsimpledb.kv.KVStore;
015import org.jsimpledb.kv.KVTransaction;
016import org.jsimpledb.kv.KVTransactionException;
017import org.jsimpledb.kv.RetryTransactionException;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * Multi-version conccurrency control (MVCC) database using snapshot views of an underlying {@link AtomicKVStore}.
023 *
024 * <p>
025 * This class provides a {@link KVDatabase} view of an underlying {@link AtomicKVStore},
026 * supporting concurrent transactions via MVCC and linearizable ACID semantics.
027 * </p>
028 *
029 * <p>
030 * Instances implement a simple optimistic locking scheme for MVCC using {@link AtomicKVStore#snapshot}. Concurrent transactions
031 * do not contend for any locks until commit time. During each transaction, reads are noted and derive from the snapshot,
032 * while writes are batched up. At commit time, if any other transaction has committed writes since the transaction's
033 * snapshot was created, and any of those writes {@linkplain Reads#isConflict conflict} with any of the committing
034 * transaction's reads, a {@link RetryTransactionException} is thrown. Otherwise, the transaction is committed and its
035 * writes are applied.
036 * </p>
037 *
038 * <p>
039 * Each outstanding transaction's mutations are batched up in memory using a {@link Writes} instance. Therefore, the
040 * transaction load supported by this class is limited to what can fit in memory.
041 * </p>
042 */
043public class SnapshotKVDatabase implements KVDatabase {
044
045// Locking order: (1) SnapshotKVTransaction, (2) SnapshotKVDatabase
046
047    protected final Logger log = LoggerFactory.getLogger(this.getClass());
048
049    private final TreeMap<Long, SnapshotVersion> versionInfoMap = new TreeMap<>();
050
051    private AtomicKVStore kvstore;
052    private long currentVersion;
053
054// Constructors
055
056    /**
057     * Default constructor.
058     *
059     * <p>
060     * The underlying key/value store must still be configured before creating any transactions.
061     */
062    public SnapshotKVDatabase() {
063    }
064
065    /**
066     * Constructor.
067     *
068     * @param kvstore underlying key/value store
069     */
070    public SnapshotKVDatabase(AtomicKVStore kvstore) {
071        this.kvstore = kvstore;
072    }
073
074// Properties
075
076    /**
077     * Get the underlying {@link AtomicKVStore}.
078     *
079     * @return underlying key/value store
080     */
081    protected synchronized AtomicKVStore getKVStore() {
082        return this.kvstore;
083    }
084
085    /**
086     * Configure the underlying {@link AtomicKVStore}.
087     *
088     * <p>
089     * Required property; must be configured before creating any transactions.
090     *
091     * @param kvstore underlying key/value store
092     * @throws IllegalStateException if there are any transactions open
093     */
094    protected synchronized void setKVStore(AtomicKVStore kvstore) {
095        if (!this.versionInfoMap.isEmpty())
096            throw new IllegalStateException("transactions exist");
097        this.kvstore = kvstore;
098    }
099
100// KVDatabase
101
102    /**
103     * Create a new transaction.
104     *
105     * @throws IllegalStateException if no {@link AtomicKVStore} is configured
106     */
107    @Override
108    public synchronized KVTransaction createTransaction() {
109
110        // Sanity check
111        if (this.kvstore == null)
112            throw new IllegalStateException("no KVSTore configured");
113
114        // Get info for the current version
115        final SnapshotVersion versionInfo = this.getCurrentSnapshotVersion();
116
117        // Create the new transaction and associate it with the current version
118        final SnapshotKVTransaction tx = this.createSnapshotKVTransaction(versionInfo);
119        versionInfo.addOpenTransaction(tx);
120        if (this.log.isDebugEnabled())
121            this.log.debug("created new transaction " + tx);
122        if (this.log.isTraceEnabled())
123            this.log.trace("updated current version info: " + versionInfo);
124
125        // Done
126        return tx;
127    }
128
129// Subclass methods
130
131    /**
132     * Instantiate a new {@link SnapshotKVTransaction} instance.
133     *
134     * <p>
135     * The implementation in {@link SnapshotKVDatabase} just invokes the {@link SnapshotKVTransaction}
136     * constructor using {@code this} and {@code versionInfo}. Subclasses may want to override this method
137     * to create a more specific subclass.
138     *
139     * @param versionInfo associated snapshot info
140     * @return new transaction instance
141     * @throws KVTransactionException if an error occurs
142     */
143    protected SnapshotKVTransaction createSnapshotKVTransaction(SnapshotVersion versionInfo) {
144        return new SnapshotKVTransaction(this, versionInfo);
145    }
146
147    /**
148     * Forcibly close all outstanding transactions.
149     *
150     * <p>
151     * Can be used by subclasses during the shutdown sequence to ensure everything is properly cleaned up.
152     * To avoid a possible lock order reversal deadlock, this instance should <b>not</b> be locked when invoking this method.
153     */
154    protected void closeTransactions() {
155
156        // Grab all remaining open transactions
157        final ArrayList<SnapshotKVTransaction> openTransactions = new ArrayList<>();
158        synchronized (this) {
159            for (SnapshotVersion versionInfo : this.versionInfoMap.values())
160                openTransactions.addAll(versionInfo.getOpenTransactions());
161        }
162
163        // Close them (but not while holding my lock, to avoid lock order reversal)
164        for (SnapshotKVTransaction tx : openTransactions) {
165            try {
166                tx.rollback();
167            } catch (Throwable e) {
168                this.log.debug("caught exception closing open transaction during shutdown (ignoring)", e);
169            }
170        }
171    }
172
173    /**
174     * Log specified exception. Used to ensure exceptions are logged at the point they are thrown.
175     *
176     * @param e exception to log
177     * @return e
178     */
179    protected KVTransactionException logException(KVTransactionException e) {
180        if (this.log.isDebugEnabled())
181            this.log.debug("throwing exception for " + e.getTransaction() + ": " + e);
182        return e;
183    }
184
185    /**
186     * Wrap a {@link RuntimeException} as needed.
187     *
188     * <p>
189     * The implementation in {@link SnapshotKVDatabase} just returns {@code e}.
190     *
191     * @param tx transaction in which the exception occurred
192     * @param e original exception
193     * @return wrapped exception, or just {@code e}
194     */
195    protected RuntimeException wrapException(SnapshotKVTransaction tx, RuntimeException e) {
196        return e;
197    }
198
199// Package methods
200
201    /**
202     * Commit a transaction.
203     */
204    synchronized void commit(SnapshotKVTransaction tx) {
205        try {
206            this.doCommit(tx);
207        } finally {
208            this.cleanupTransaction(tx);
209        }
210    }
211
212    /**
213     * Rollback a transaction.
214     */
215    synchronized void rollback(SnapshotKVTransaction tx) {
216        if (this.log.isDebugEnabled())
217            this.log.debug("rolling back transaction " + tx);
218        this.cleanupTransaction(tx);
219    }
220
221// Internal methods
222
223    private synchronized void doCommit(SnapshotKVTransaction tx) {
224
225        // Get current and transaction's version info
226        final SnapshotVersion currentSnapshotVersion = this.getCurrentSnapshotVersion();
227        final SnapshotVersion transactionSnapshotVersion = tx.getSnapshotVersion();
228        final long transactionVersion = transactionSnapshotVersion.getVersion();
229        assert this.currentVersion - transactionVersion >= 0;
230        assert transactionSnapshotVersion.getOpenTransactions().contains(tx);
231
232        // Debug
233        if (this.log.isDebugEnabled()) {
234            this.log.debug("committing transaction " + tx + " based on version "
235              + transactionVersion + " (current version is " + this.currentVersion + ")");
236        }
237
238        // Check whether transaction has been forcibly killed somehow
239        if (!transactionSnapshotVersion.getOpenTransactions().contains(tx))
240            throw this.logException(new RetryTransactionException(tx, "transaction has been forcibly invalidated"));
241
242        // Get transaction reads & writes
243        final Reads transactionReads = tx.getMutableView().getReads();
244        final Writes transactionWrites = tx.getMutableView().getWrites();
245
246        // If the current version has advanced past the transaction's version, check for conflicts from intervening commits
247        for (long version = transactionVersion; version != this.currentVersion; version++) {
248            final SnapshotVersion committedSnapshotVersion = this.versionInfoMap.get(version);
249            final Writes committedWrites = committedSnapshotVersion.getCommittedWrites();
250            final boolean conflict = transactionReads.isConflict(committedWrites);
251            if (this.log.isDebugEnabled()) {
252                this.log.debug("ordering " + tx + " after writes in version " + version + " results in "
253                  + (conflict ? "conflict" : "no conflict"));
254                if (this.log.isTraceEnabled())
255                    this.log.trace("transaction reads: {} committed writes: {}", transactionReads, committedWrites);
256            }
257            if (conflict) {
258                throw this.logException(new RetryTransactionException(tx, "transaction is based on MVCC version "
259                  + transactionSnapshotVersion.getVersion() + " but the transaction committed at MVCC version "
260                  + version + " contains conflicting writes"));
261            }
262        }
263
264        // Atomically apply the transaction's mutations
265        if (this.log.isDebugEnabled())
266            this.log.debug("applying mutations of " + tx + " to SnapshotMVCC database");
267        this.kvstore.mutate(transactionWrites, true);
268
269        // Record transaction's writes for this version
270        currentSnapshotVersion.setCommittedWrites(transactionWrites);
271
272        // Advance to the next MVCC version
273        if (this.log.isDebugEnabled())
274            this.log.debug("updating current version from " + this.currentVersion + " -> " + (this.currentVersion + 1));
275        this.currentVersion++;
276    }
277
278    private void cleanupTransaction(SnapshotKVTransaction tx) {
279
280        // Debug
281        assert Thread.holdsLock(this);
282        if (this.log.isTraceEnabled())
283            this.log.trace("cleaning up transaction " + tx);
284
285        // Remove open transaction from version
286        tx.getSnapshotVersion().removeOpenTransaction(tx);
287
288        // Discard all versions older than all remaining open transactions
289        for (Iterator<Map.Entry<Long, SnapshotVersion>> i = this.versionInfoMap.entrySet().iterator(); i.hasNext(); ) {
290            final SnapshotVersion versionInfo = i.next().getValue();
291            if (!versionInfo.getOpenTransactions().isEmpty())
292                break;
293            if (this.log.isDebugEnabled())
294                this.log.debug("discarding obsolete version " + versionInfo);
295            final KVStore snapshot = versionInfo.getSnapshot();
296            if (snapshot instanceof AutoCloseable) {
297                try {
298                    ((AutoCloseable)snapshot).close();
299                } catch (Exception e) {
300                    // ignore
301                }
302            }
303            i.remove();
304        }
305    }
306
307    // Get SnapshotVersion for the current MVCC version, creating on demand if necessary
308    private SnapshotVersion getCurrentSnapshotVersion() {
309        SnapshotVersion versionInfo = this.versionInfoMap.get(this.currentVersion);
310        if (versionInfo == null) {
311            versionInfo = new SnapshotVersion(this.currentVersion, this.kvstore.snapshot());
312            this.versionInfoMap.put(this.currentVersion, versionInfo);
313            if (this.log.isTraceEnabled())
314                this.log.trace("created new version " + versionInfo);
315        }
316        return versionInfo;
317    }
318}
