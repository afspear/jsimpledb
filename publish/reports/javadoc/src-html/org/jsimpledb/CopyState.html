006package org.jsimpledb;
007
008import com.google.common.primitives.Ints;
009
010import java.util.Arrays;
011import java.util.Map;
012import java.util.TreeMap;
013
014import org.jsimpledb.core.ObjId;
015import org.jsimpledb.util.ObjIdSet;
016
017/**
018 * Keeps tracks of which objects have already been copied when copying objects between transactions.
019 *
020 * @see JObject#copyTo JObject.copyTo()
021 * @see JTransaction#copyTo(JTransaction, JObject, ObjId, CopyState, String...) JTransaction.copyTo()
022 */
023public class CopyState implements Cloneable {
024
025    private final TreeMap<int[], ObjIdSet> traversedMap = new TreeMap<>(Ints.lexicographicalComparator());
026    private /*final*/ ObjIdSet copied;
027
028    /**
029     * Default constructor.
030     */
031    public CopyState() {
032        this(new ObjIdSet());
033    }
034
035    /**
036     * Constructor to use when a set of known objects has already been copied.
037     *
038     * @param copied the ID's of objects that have already been copied
039     * @throws IllegalArgumentException if {@code copied} is null
040     */
041    public CopyState(ObjIdSet copied) {
042        if (copied == null)
043            throw new IllegalArgumentException("null copied");
044        this.copied = copied.clone();
045        synchronized (this) { }
046    }
047
048    /**
049     * Determine if an object has already been copied, and if not mark it so.
050     *
051     * @param id object ID of object being copied
052     * @return true if {@code id} was not previously marked copied, otherwise false
053     * @throws IllegalArgumentException if {@code id} is null
054     */
055    public boolean markCopied(ObjId id) {
056        if (id == null)
057            throw new IllegalArgumentException("null id");
058        return this.copied.add(id);
059    }
060
061    /**
062     * Determine if the specified reference path has already been traversed starting at the given object, and if not mark it so.
063     *
064     * @param id object ID of object being copied
065     * @param fields reference path storage IDs
066     * @return true if {@code fields} was not previously marked as traversed from {@code id}, otherwise false
067     * @throws IllegalArgumentException if either parameter is null
068     * @throws IllegalArgumentException if {@code fields} has length zero
069     */
070    public boolean markTraversed(ObjId id, int[] fields) {
071
072        // Sanity check
073        if (id == null)
074            throw new IllegalArgumentException("null id");
075        if (fields == null)
076            throw new IllegalArgumentException("null fields");
077        if (fields.length == 0)
078            throw new IllegalArgumentException("empty fields");
079
080        // Mark the path, and all prefixes of the path, as having been traversed
081        boolean fullPath = true;
082        for (int limit = fields.length; limit > 0; limit--) {
083            final int[] prefix = fullPath ? fields : Arrays.copyOfRange(fields, 0, limit);
084            ObjIdSet idSet = this.traversedMap.get(prefix);
085            if (idSet == null) {
086                idSet = new ObjIdSet();
087                this.traversedMap.put(prefix, idSet);
088            }
089            if (!idSet.add(id))                         // we have already marked this prefix (and every shorter prefix)
090                return !fullPath;
091            fullPath = false;
092        }
093
094        // Done
095        return true;
096    }
097
098// Cloneable
099
100    @Override
101    public CopyState clone() {
102        final CopyState clone;
103        try {
104            clone = (CopyState)super.clone();
105        } catch (CloneNotSupportedException e) {
106            throw new RuntimeException(e);
107        }
108        for (Map.Entry<int[], ObjIdSet> entry : this.traversedMap.entrySet())
109            clone.traversedMap.put(entry.getKey(), entry.getValue().clone());
110        clone.copied = this.copied.clone();
111        return clone;
112    }
113}
