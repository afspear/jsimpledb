006package org.jsimpledb.parse.expr;
007
008import com.google.common.base.Function;
009import com.google.common.collect.Lists;
010
011import java.lang.reflect.InvocationTargetException;
012import java.lang.reflect.Method;
013import java.util.LinkedHashSet;
014import java.util.List;
015import java.util.Set;
016
017import org.jsimpledb.parse.ParseSession;
018
019/**
020 * {@link Node} that invokes a Java method when evaluated.
021 */
022public class MethodInvokeNode implements Node {
023
024    private final Class<?> klass;
025    private final Node node;
026    private final String name;
027    private final List<Node> paramNodes;
028
029    /**
030     * Constructor for static method invocation.
031     *
032     * @param klass class containing static method
033     * @param name static method name
034     * @param paramNodes method parameters
035     */
036    public MethodInvokeNode(Class<?> klass, String name, List<Node> paramNodes) {
037        this(name, paramNodes, klass, null);
038        if (klass == null)
039            throw new IllegalArgumentException("null klass");
040    }
041
042    /**
043     * Constructor for instance method invocation.
044     *
045     * @param node node evaluating to target object
046     * @param name instance method name
047     * @param paramNodes method parameters
048     */
049    public MethodInvokeNode(Node node, String name, List<Node> paramNodes) {
050        this(name, paramNodes, null, node);
051        if (node == null)
052            throw new IllegalArgumentException("null node");
053    }
054
055    private MethodInvokeNode(String name, List<Node> paramNodes, Class<?> klass, Node node) {
056        if (name == null)
057            throw new IllegalArgumentException("null target");
058        if (paramNodes == null)
059            throw new IllegalArgumentException("null paramNodes");
060        for (Node paramNode : paramNodes) {
061            if (paramNode == null)
062                throw new IllegalArgumentException("null paramNode in list");
063        }
064        this.klass = klass;
065        this.node = node;
066        this.name = name;
067        this.paramNodes = paramNodes;
068    }
069
070    @Override
071    public Value evaluate(final ParseSession session) {
072
073        // Evaluate params
074        final Object[] params = Lists.transform(paramNodes, new Function<Node, Object>() {
075            @Override
076            public Object apply(Node param) {
077                return param.evaluate(session).get(session);
078            }
079        }).toArray();
080
081        // Handle static method
082        if (this.klass != null)
083            return this.invokeMethod(this.klass, null, name, params);
084
085        // Handle instance method
086        final Object obj = this.node.evaluate(session).checkNotNull(session, "method " + name + "() invocation");
087        return this.invokeMethod(obj.getClass(), obj, name, params);
088    }
089
090    private Value invokeMethod(Class<?> cl, Object obj, String name, Object[] params) {
091
092        // Try interface methods
093        for (Class<?> iface : this.addInterfaces(cl, new LinkedHashSet<Class<?>>())) {
094            for (Method method : iface.getMethods()) {
095                final Value value = this.tryMethod(method, obj, name, params);
096                if (value != null)
097                    return value;
098            }
099        }
100
101        // Try class methods
102        for (Method method : cl.getMethods()) {
103            final Value value = this.tryMethod(method, obj, name, params);
104            if (value != null)
105                return value;
106        }
107
108        // Not found
109        throw new EvalException("no compatible method `" + name + "()' found in " + cl);
110    }
111
112    private Set<Class<?>> addInterfaces(Class<?> cl, Set<Class<?>> interfaces) {
113        for (Class<?> iface : cl.getInterfaces()) {
114            interfaces.add(iface);
115            this.addInterfaces(iface, interfaces);
116        }
117        if (cl.getSuperclass() != null)
118            this.addInterfaces(cl.getSuperclass(), interfaces);
119        return interfaces;
120    }
121
122    private Value tryMethod(Method method, Object obj, String name, Object[] params) {
123        if (!method.getName().equals(name))
124            return null;
125        final Class<?>[] ptypes = method.getParameterTypes();
126        if (method.isVarArgs()) {
127            if (params.length < ptypes.length - 1)
128                return null;
129            Object[] newParams = new Object[ptypes.length];
130            System.arraycopy(params, 0, newParams, 0, ptypes.length - 1);
131            Object[] varargs = new Object[params.length - (ptypes.length - 1)];
132            System.arraycopy(params, ptypes.length - 1, varargs, 0, varargs.length);
133            newParams[ptypes.length - 1] = varargs;
134            params = newParams;
135        } else if (params.length != ptypes.length)
136            return null;
137        final Object result;
138        try {
139            result = method.invoke(obj, params);
140        } catch (IllegalArgumentException e) {
141            return null;                            // a parameter type didn't match -> wrong method
142        } catch (Exception e) {
143            final Throwable t = e instanceof InvocationTargetException ?
144              ((InvocationTargetException)e).getTargetException() : e;
145            throw new EvalException("error invoking method `" + name + "()' on "
146              + (obj != null ? "object of type " + obj.getClass().getName() : method.getDeclaringClass()) + ": " + t, t);
147        }
148        return result != null || method.getReturnType() != Void.TYPE ? new ConstValue(result) : Value.NO_VALUE;
149    }
150}
