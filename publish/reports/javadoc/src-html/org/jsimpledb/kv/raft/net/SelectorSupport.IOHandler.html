006package org.jsimpledb.kv.raft.net;
007
008import com.google.common.base.Preconditions;
009
010import java.io.IOException;
011import java.nio.channels.ClosedChannelException;
012import java.nio.channels.SelectableChannel;
013import java.nio.channels.SelectionKey;
014import java.nio.channels.Selector;
015import java.util.ArrayList;
016import java.util.Iterator;
017
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * Support superclass for classes that perform NIO selection operations.
023 */
024public abstract class SelectorSupport {
025
026    protected final Logger log = LoggerFactory.getLogger(this.getClass());
027
028    /**
029     * Create a new {@link SelectionKey} by registering the given channel on the given {@link Selector}
030     * and associating the specified {@link IOHandler} to handle I/O ready conditions.
031     *
032     * <p>
033     * Initially, no I/O operations will be selected. Use {@link #selectFor selectFor()} to add/remove them.
034     *
035     * @param selector NIO selector
036     * @param channel I/O channel
037     * @param handler I/O handler
038     * @return new selection key
039     * @throws IllegalArgumentException if any parameter is null
040     * @throws ClosedChannelException if {@code channel} is closed
041     */
042    protected SelectionKey createSelectionKey(Selector selector, SelectableChannel channel, IOHandler handler)
043      throws ClosedChannelException {
044
045        // Sanity check
046        Preconditions.checkArgument(selector != null, "null selector");
047        Preconditions.checkArgument(channel != null, "null channel");
048        Preconditions.checkArgument(handler != null, "null handler");
049
050        // Register channel with our selector
051        return channel.register(selector, 0, handler);
052    }
053
054    /**
055     * Enable or disable listening for the specified I/O operation(s).
056     *
057     * @param selectionKey selection key; may be null, in which case this method does nothing
058     * @param ops I/O operations
059     * @param enabled true to enable, false to disable
060     */
061    protected void selectFor(SelectionKey selectionKey, int ops, boolean enabled) {
062        if (selectionKey == null)
063            return;
064        final int currentOps = selectionKey.interestOps();
065        selectionKey.interestOps(enabled ? currentOps | ops : currentOps & ~ops);
066    }
067
068    /**
069     * Handle any ready I/O operations available from the given {@link Selector}.
070     *
071     * <p>
072     * This assumes an {@link IOHandler} is assocated with each {@link SelectionKey} (e.g., they were created
073     * via {@link #createSelectionKey createSelectionKey()}).
074     *
075     * @param selector NIO selector
076     */
077    protected void handleIO(Selector selector) {
078        for (Iterator<SelectionKey> i = selector.selectedKeys().iterator(); i.hasNext(); ) {
079            final SelectionKey key = i.next();
080            i.remove();
081            final IOHandler handler = (IOHandler)key.attachment();
082            if (this.log.isTraceEnabled())
083                this.log.trace("I/O ready: key=" + dbg(key) + " handler=" + handler);
084            try {
085                handler.serviceIO(key);
086            } catch (IOException e) {
087                if (this.log.isDebugEnabled())
088                    this.log.debug("I/O error from " + handler, e);
089                handler.close(e);
090            } catch (Throwable t) {
091                this.log.error("service error from " + handler, t);
092                handler.close(t);
093            }
094        }
095    }
096
097    /**
098     * Get a debug description of the given keys.
099     *
100     * @param keys selection keys
101     * @return debug description
102     */
103    protected static String dbg(Iterable<? extends SelectionKey> keys) {
104        final ArrayList<String> strings = new ArrayList<>();
105        for (SelectionKey key : keys)
106            strings.add(dbg(key));
107        return strings.toString();
108    }
109
110    /**
111     * Get a debug description of the given key.
112     *
113     * @param key selection key
114     * @return debug description
115     */
116    protected static String dbg(SelectionKey key) {
117        try {
118            return "Key[interest=" + dbgOps(key.interestOps()) + ",ready="
119              + dbgOps(key.readyOps()) + ",obj=" + key.attachment() + "]";
120        } catch (java.nio.channels.CancelledKeyException e) {
121            return "Key[canceled]";
122        }
123    }
124
125    /**
126     * Get a debug description of the given I/O ops.
127     *
128     * @param ops I/O operation bits
129     * @return debug description
130     */
131    protected static String dbgOps(int ops) {
132        final StringBuilder buf = new StringBuilder(4);
133        if ((ops & SelectionKey.OP_ACCEPT) != 0)
134            buf.append("A");
135        if ((ops & SelectionKey.OP_CONNECT) != 0)
136            buf.append("C");
137        if ((ops & SelectionKey.OP_READ) != 0)
138            buf.append("R");
139        if ((ops & SelectionKey.OP_WRITE) != 0)
140            buf.append("W");
141        return buf.toString();
142    }
143
144// IOHandler
145
146    /**
147     * Callback interface used by {@link SelectorSupport}.
148     */
149    public interface IOHandler {
150
151        /**
152         * Handle ready I/O.
153         *
154         * @param key selection key
155         * @throws IOException if an I/O error occurs; this will result in this instance being {@link #close close()}'ed
156         */
157        void serviceIO(SelectionKey key) throws IOException;
158
159        /**
160         * Notification that an exception was thrown by {@link #serviceIO serviceIO()}.
161         * Typically this method will close the associated channel, which implicitly unregisters
162         * the associated {@link SelectionKey}s.
163         *
164         * <p>
165         * This method must be idempotent.
166         *
167         * @param cause the error that occurred
168         */
169        void close(Throwable cause);
170    }
171}
