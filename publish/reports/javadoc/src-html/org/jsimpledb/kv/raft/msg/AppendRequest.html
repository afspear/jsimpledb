006package org.jsimpledb.kv.raft.msg;
007
008import com.google.common.base.Preconditions;
009
010import java.nio.ByteBuffer;
011
012import org.jsimpledb.kv.raft.Timestamp;
013import org.jsimpledb.util.LongEncoder;
014
015/**
016 * Sent from leaders to followers to probe the follower's log state and/or append an entry to their log.
017 *
018 * <p>
019 * Instances also provide the {@linkplain #getLeaderLeaseTimeout leader's lease timeout} value, which is
020 * used to commit read-only transactions, as well as a {@linkplain #getLeaderTimestamp leader timestamp}
021 * which should be reflected back in the corresponding {@link AppendResponse}.
022 */
023public class AppendRequest extends Message {
024
025    private final Timestamp leaderTimestamp;        // leader's timestamp for this request
026    private final Timestamp leaderLeaseTimeout;     // earliest leader timestamp at which time leader could be deposed
027    private final long leaderCommit;                // index of highest log entry known to be committed
028    private final long prevLogTerm;                 // term of previous log entry
029    private final long prevLogIndex;                // index of previous log entry
030    private final long logEntryTerm;                // term corresponding to log entry, or zero if this is a "probe"
031
032    private ByteBuffer mutationData;                // serialized mutations, if not a probe and not from follower transaction
033    private boolean mutationDataInvalid;            // mutationData has already been grabbed
034
035// Constructors
036
037    /**
038     * Construtor for a "probe" that does not contain a log entry.
039     *
040     * @param clusterId cluster ID
041     * @param senderId identity of sender
042     * @param recipientId identity of recipient
043     * @param term sender's current term
044     * @param leaderTimestamp leader's timestamp for this request
045     * @param leaderLeaseTimeout earliest leader timestamp at which leader could be deposed
046     * @param leaderCommit current commit index for sender
047     * @param prevLogTerm term of the log entry just prior to this one
048     * @param prevLogIndex index of the log entry just prior to this one
049     */
050    public AppendRequest(int clusterId, String senderId, String recipientId, long term,
051      Timestamp leaderTimestamp, Timestamp leaderLeaseTimeout, long leaderCommit, long prevLogTerm, long prevLogIndex) {
052        this(clusterId, senderId, recipientId, term, leaderTimestamp, leaderLeaseTimeout, leaderCommit,
053          prevLogTerm, prevLogIndex, 0, null);
054    }
055
056    /**
057     * Constructor for a request that contains an actual log entry.
058     *
059     * @param clusterId cluster ID
060     * @param senderId identity of sender
061     * @param recipientId identity of recipient
062     * @param term sender's current term
063     * @param leaderTimestamp leader's timestamp for this request
064     * @param leaderLeaseTimeout earliest leader timestamp at which leader could be deposed
065     * @param leaderCommit current commit index for sender
066     * @param prevLogTerm term of the log entry just prior to this one
067     * @param prevLogIndex index of the log entry just prior to this one
068     * @param logEntryTerm term of this log entry
069     * @param mutationData log entry serialized mutations, or null if follower should have the data already
070     */
071    public AppendRequest(int clusterId, String senderId, String recipientId, long term, Timestamp leaderTimestamp,
072      Timestamp leaderLeaseTimeout, long leaderCommit, long prevLogTerm, long prevLogIndex, long logEntryTerm,
073      ByteBuffer mutationData) {
074        super(Message.APPEND_REQUEST_TYPE, clusterId, senderId, recipientId, term);
075        this.leaderTimestamp = leaderTimestamp;
076        this.leaderLeaseTimeout = leaderLeaseTimeout;
077        this.leaderCommit = leaderCommit;
078        this.prevLogTerm = prevLogTerm;
079        this.prevLogIndex = prevLogIndex;
080        this.logEntryTerm = logEntryTerm;
081        this.mutationData = mutationData;
082        this.checkArguments();
083    }
084
085    AppendRequest(ByteBuffer buf) {
086        super(Message.APPEND_REQUEST_TYPE, buf);
087        this.leaderTimestamp = Message.getTimestamp(buf);
088        this.leaderLeaseTimeout = this.leaderTimestamp.offset((int)LongEncoder.read(buf));
089        this.leaderCommit = LongEncoder.read(buf);
090        this.prevLogTerm = LongEncoder.read(buf);
091        this.prevLogIndex = LongEncoder.read(buf);
092        this.logEntryTerm = LongEncoder.read(buf);
093        this.mutationData = this.logEntryTerm != 0 && Message.getBoolean(buf) ? Message.getByteBuffer(buf) : null;
094        this.checkArguments();
095    }
096
097    @Override
098    void checkArguments() {
099        super.checkArguments();
100        Preconditions.checkArgument(this.leaderTimestamp != null);
101        Preconditions.checkArgument(this.leaderLeaseTimeout != null);
102        Preconditions.checkArgument(this.leaderCommit >= 0);
103        Preconditions.checkArgument(this.prevLogTerm >= 0);
104        Preconditions.checkArgument(this.prevLogIndex >= 0);
105        Preconditions.checkArgument(this.logEntryTerm >= 0);
106        Preconditions.checkArgument(this.mutationData == null || this.logEntryTerm > 0);
107    }
108
109// Properties
110
111    public Timestamp getLeaderTimestamp() {
112        return this.leaderTimestamp;
113    }
114
115    public Timestamp getLeaderLeaseTimeout() {
116        return this.leaderLeaseTimeout;
117    }
118
119    public long getLeaderCommit() {
120        return this.leaderCommit;
121    }
122
123    public long getPrevLogTerm() {
124        return this.prevLogTerm;
125    }
126
127    public long getPrevLogIndex() {
128        return this.prevLogIndex;
129    }
130
131    public boolean isProbe() {
132        return this.logEntryTerm == 0;
133    }
134
135    public long getLogEntryTerm() {
136        return this.logEntryTerm;
137    }
138
139    /**
140     * Get the serialized data for the log entry, if any.
141     * Returns null if this is a probe or follower is expected to already have the data from a transaction.
142     *
143     * <p>
144     * This method may only be invoked once.
145     *
146     * @return log entry serialized mutations, or null if this message does not contain data
147     * @throws IllegalStateException if this method has already been invoked
148     */
149    public ByteBuffer getMutationData() {
150        Preconditions.checkState(!this.mutationDataInvalid);
151        final ByteBuffer result = this.mutationData;
152        this.mutationData = null;
153        this.mutationDataInvalid = true;
154        return result;
155    }
156
157// Message
158
159    @Override
160    public boolean isLeaderMessage() {
161        return true;
162    }
163
164    @Override
165    public void visit(MessageSwitch handler) {
166        handler.caseAppendRequest(this);
167    }
168
169    @Override
170    public void writeTo(ByteBuffer dest) {
171        Preconditions.checkState(!this.mutationDataInvalid);
172        super.writeTo(dest);
173        Message.putTimestamp(dest, this.leaderTimestamp);
174        LongEncoder.write(dest, this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp));
175        LongEncoder.write(dest, this.leaderCommit);
176        LongEncoder.write(dest, this.prevLogTerm);
177        LongEncoder.write(dest, this.prevLogIndex);
178        LongEncoder.write(dest, this.logEntryTerm);
179        if (this.logEntryTerm != 0) {
180            Message.putBoolean(dest, this.mutationData != null);
181            if (this.mutationData != null)
182                Message.putByteBuffer(dest, this.mutationData);
183        }
184    }
185
186    @Override
187    protected int calculateSize() {
188        Preconditions.checkState(!this.mutationDataInvalid);
189        return super.calculateSize()
190          + Message.calculateSize(this.leaderTimestamp)
191          + LongEncoder.encodeLength(this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp))
192          + LongEncoder.encodeLength(this.leaderCommit)
193          + LongEncoder.encodeLength(this.prevLogTerm)
194          + LongEncoder.encodeLength(this.prevLogIndex)
195          + LongEncoder.encodeLength(this.logEntryTerm)
196          + (this.logEntryTerm != 0 ? 1 + (this.mutationData != null ? Message.calculateSize(this.mutationData) : 0) : 0);
197    }
198
199// Object
200
201    @Override
202    public String toString() {
203        return this.getClass().getSimpleName()
204          + "[\"" + this.getSenderId() + "\"->\"" + this.getRecipientId() + "\""
205          + ",clusterId=" + String.format("%08x", this.getClusterId())
206          + ",term=" + this.getTerm()
207          + ",leaderTimestamp=" + this.leaderTimestamp
208          + ",leaderLeaseTimeout=" + String.format("%+dms", this.leaderLeaseTimeout.offsetFrom(this.leaderTimestamp))
209          + ",leaderCommit=" + this.leaderCommit
210          + ",prevLog=" + this.prevLogIndex + "t" + this.prevLogTerm
211          + (this.logEntryTerm != 0 ? ",logEntryTerm=" + this.logEntryTerm : "")
212          + (this.mutationData != null ?
213            ",mutationData=" + this.describe(this.mutationData) : this.mutationDataInvalid ? ",mutationData=invalid" : "")
214          + "]";
215    }
216}
